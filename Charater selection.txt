        break;

        document.getElementById("name").innerHTML = "Shokuhou Misaki";
        break;
    case "hayasaka":
        document.getElementById("img").src = "./HMTL project images/hayasaka render.png";
        document.getElementById("name").innerHTML = "Hayasaka Ai";
